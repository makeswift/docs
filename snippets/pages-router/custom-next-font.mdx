import type { AppProps } from 'next/app'
import localFont from 'next/font/local'
import '../styles/globals.css'

const yourBrand = localFont({
  src: [
    { path: '../public/fonts/YourBrand-Regular.woff2', weight: '400', style: 'normal' },
    { path: '../public/fonts/YourBrand-Bold.woff2',    weight: '700', style: 'normal' },
  ],
  display: 'swap',
})

export default function MyApp({ Component, pageProps }: AppProps) {
  return (
    <main className={yourBrand.className}>
      <Component {...pageProps} />
    </main>
  )
}
